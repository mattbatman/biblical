    2
  </span>

  <span class="verse" id="1SA26V3">
    3
  </span>

  <span class="verse" id="1SA26V4">
    4
  </span>

  <span class="verse" id="1SA26V5">
    5
  </span>

<div class="p">
  <span class="verse" id="1SA26V6">

    8
  </span>

  <span class="verse" id="1SA26V9">
    9

  <span class="verse" id="1SA26V10">
    10
  </span>

    11

  <span class="verse" id="1SA26V12">
    12
  </span>

  <span class="verse" id="1SA26V13">
    13
  </span>

    14

  <span class="verse" id="1SA26V15">
    15

  <span class="verse" id="1SA26V16">
    16

</div>

    20
